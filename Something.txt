# create_project.py
import os
from datetime import datetime

base = os.path.expanduser(r"~\Projects")  # change if needed
proj = input("Enter Project Name: ").strip()
if not proj:
    print("No name entered.")
    raise SystemExit

project_folder = os.path.join(base, proj)
os.makedirs(project_folder, exist_ok=True)

checklist_path = os.path.join(project_folder, "CHECKLIST.md")
content = f"""# {proj} - Workflow Checklist

## Steps
- [ ] Send BU Notification
- [ ] Send GSOC Notification
- [ ] Change status in ServiceNow (SNOW)
- [ ] Work on the task
- [ ] Send End Notification
- [ ] Send Report
- [ ] Send Flaw Data
- [ ] Change status in ServiceNow (SNOW) - Final Update

## Status Tracking
- Start Date: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}
- End Date:

## Notes
(Add comments here)
"""
with open(checklist_path, "w", encoding="utf-8") as f:
    f.write(content)

print("Checklist created at:", checklist_path)
